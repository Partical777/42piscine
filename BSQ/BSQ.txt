#include <stdio.h>
#include <unistd.h>

int main(void) {
  int i;
  int j;

  const char arr[9][27] = {
    "............ooo............",
    "....o....o.........oo...o..", 
    ".....ooo....o...oo........", 
    "ooo.............oo.........", 
    "....o...........oo.........", 
    ".......ooooo...o.......o...", 
    ".oo........................", 
    "......o....................", 
    "..o.......o................"
  };
  i = 0;
  j = 0;
  for(i = 0 ; i < 9 ; i++){
    for(j = 0 ; j < 27 ; j++){
      printf("%c", arr[i][j]);
    }
    printf("\n");
  }

  int linenumber = 9; //行數
  int biggestsofar = 0; //目前找到最大的
  int obstacle = 0;
  int pointX = 0; //目前找到最大的點x值
  int pointY = 0; //目前找到最大的點y值

  for (int x = 0 ; x < (linenumber - biggestsofar) ; x++){
    for (int y = 0 ; y < (27 - biggestsofar) ; y++){

      for (int testbiggest = (linenumber - x) ; testbiggest > biggestsofar ; testbiggest--){
        obstacle = 0;

          for (int testx = 0 ; testx < testbiggest ; testx++){
              for (int testy = 0 ; testy < testbiggest ; testy++){
                if (arr[x + testx][y + testy] == 'o'){
                  obstacle = 1;
                  break;
                }
              }
              if (obstacle == 1){
                break;
              }
          }
          if (obstacle == 0){
            pointX = x;
            pointY = y;
            biggestsofar = testbiggest;
            break;
          }
        
      }

    }
  }
  printf("%d\n", biggestsofar);
  printf("[%d, %d]", pointX, pointY);
  
  return 0;
}
