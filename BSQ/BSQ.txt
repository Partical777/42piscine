#include <stdio.h>
#include <unistd.h>

int main(void) {
  int i;
  int j;

  const char arr[9][27] = {
    "...........................",
    "....o......................", 
    "............o..............", 
    "...........................", 
    "....o......................", 
    "...............o...........", 
    "...........................", 
    "......o..............o.....", 
    "..o.......o................"
  };
  i = 0;
  j = 0;
  for(i = 0 ; i < 9 ; i++){
    for(j = 0 ; j < 27 ; j++){
      printf("%c", arr[i][j]);
    }
    printf("\n");
  }
  // while (i < 9){
  //   while (j < 27){
  //     write(1, arr[0][j], 1);

  //     j++;
  //   }
  //   j = 0;
  //   write(1, "\n", 1);
  //   i++;
  // }
  int linenumber = 9; //行數
  int biggestsofar = 0;
  int obstacle = 0;
  int pointX = 0;
  int pointY = 0;

  for (int x = 0 ; x < (linenumber - biggestsofar) ; x++){
    for (int y = 0 ; y < 27 ; y++){

      for (int testbiggest = linenumber ; testbiggest > biggestsofar ; testbiggest--){
        obstacle = 0;

          for (int testx = 0 ; testx < testbiggest ; testx++){
              for (int testy = 0 ; testy < testbiggest ; testy++){
                if (arr[x + testx][y + testy] == 'o'){
                  obstacle = 1;
                  break;
                }
              }
              if (obstacle == 1){
                break;
              }
          }
          if (obstacle == 0){
            pointX = x;
            pointY = y;
            biggestsofar = testbiggest;
            break;
          }
        
      }

    }
  }
  printf("%d\n", biggestsofar);
  printf("[%d, %d]", pointX, pointY);
  
  return 0;
}
